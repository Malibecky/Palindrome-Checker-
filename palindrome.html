Is it a Palindrome?

Enter in text to check for a palindrome:


💡 A palindrome is a word or sentence that's spelled the same way both forward and backward, ignoring punctuation, case, and spacing.
import re from pyscript import Element def palindrome(): # get the user input text = Element("text-input").element.value.lower() # omit all special symbols and display the alphanumeric characters as an array split_text = re.findall("[a-z0-9]", text) # join the array joint_text = ''.join(split_text) # reverse the array reversed_arr = split_text[::-1] # join the reversed array reversed_text = ''.join(reversed_arr) # compare if the joint_text is same as the reversed_text if joint_text == reversed_text: result = f"{text} is a palindrome" else: result = f"{text} is not a palindrome" Element("result").element.textContent = result
